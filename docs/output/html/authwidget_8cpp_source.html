259void AuthWidget::onLoginClicked()
260{
261 if (isLocked) {
262 int remainingSec = lockTimer->remainingTime() / 1000;
263 int remainingMin = remainingSec / 60;
264 int remainingSecMod = remainingSec % 60;
265 statusLabel->setText(remainingMin > 0
266 ? QString("Заблокировано. Осталось %1 мин %2 сек").arg(remainingMin).arg(remainingSecMod)
267 : QString("Заблокировано. Осталось %1 сек").arg(remainingSec));
268 statusLabel->show();
269 return;
270 }
271
272 QString login = loginEdit->text().trimmed();
273 QString password = passwordEdit->text();
274
275 if (login.isEmpty() || password.isEmpty()) {
276 statusLabel->setText("Введите логин и пароль.");
277 statusLabel->setStyleSheet(QString("QLabel { color: %1; border: none; font-size: %2pt; }").arg(GH_RED).arg(FONT_SIZE_SMALL));
278 statusLabel->show();
279 return;
280 }
281
282 loginBtn->setEnabled(false);
283 statusLabel->setText("Подключаемся...");
284 statusLabel->setStyleSheet(QString("QLabel { color: %1; border: none; font-size: %2pt; }").arg(GH_MUTED).arg(FONT_SIZE_SMALL));
285 statusLabel->show();
286
287 QByteArray hashBytes = QCryptographicHash::hash(password.toUtf8(), QCryptographicHash::Sha256);
288 QString passwordHash = QString::fromLatin1(hashBytes.toHex());
289
290 m_waitingForAuth = true;
291 ClientSingleton::instance().sendRequestAsync(QString("auth||%1||%2").arg(login, passwordHash));
292}

294void AuthWidget::onAuthResponseReceived(const QString &response)
295{
296 if (!m_waitingForAuth) return;
297 m_waitingForAuth = false;
298
299 QString r = response.trimmed();
300 if (r.isEmpty()) {
301 loginBtn->setEnabled(true);
302 statusLabel->setText("Ошибка соединения с сервером.");
303 statusLabel->setStyleSheet(QString("QLabel { color: %1; border: none; font-size: %2pt; }").arg(GH_RED).arg(FONT_SIZE_SMALL));
304 statusLabel->show();
305 return;
306 }
307
308 if (r == "auth_code_sent") {
309 statusLabel->hide();
310 emit showVerifyAuth(loginEdit->text().trimmed());
311 return;
312 }
313
314 loginBtn->setEnabled(true);
315 statusLabel->setStyleSheet(QString("QLabel { color: %1; border: none; font-size: %2pt; }").arg(GH_RED).arg(FONT_SIZE_SMALL));
316
317 if (r == "auth-") {
318 failedAttempts++;
319 if (failedAttempts < 4) {
320 statusLabel->setText(QString("Неверный логин или пароль. Осталось попыток: %1").arg(4 - failedAttempts));
321 statusLabel->show();
322 return;
323 }
324 if (failedAttempts == 4) { lockLevel = 1; applyLock(0, "Заблокировано на 30 секунд"); return; }
325 if (failedAttempts == 5) { lockLevel = 2; applyLock(5, "Заблокировано на 5 минут"); return; }
326 if (failedAttempts == 6) { lockLevel = 3; applyLock(10, "Заблокировано на 10 минут"); return; }
327 lockLevel = 4; applyLock(9999, "Аккаунт заблокирован на длительное время");
328 return;
329 }
330
331 m_waitingForAuth = false;
332}
