88void GraphWidget::onPlotClicked()
89{
90 taskLabel->setText("Загрузка данных...");
91 taskLabel->repaint();
92
93 double xMin = xMinSpin->value();
94 double xMax = xMaxSpin->value();
95 double step = stepSpin->value();
96
97 if (xMin >= xMax) {
98 taskLabel->setText("Ошибка: X мин должен быть меньше X макс.");
99 return;
100 }
101
102 ClientSingleton &client = ClientSingleton::instance();
103 if (!client.isConnected()) {
104 taskLabel->setText("Ошибка: нет подключения к серверу.");
105 return;
106 }
107
108 QString request = QString("get_graph||%1||%2||%3")
109 .arg(xMin, 0, 'f', 6)
110 .arg(xMax, 0, 'f', 6)
111 .arg(step, 0, 'f', 6);
112
113 QString response = client.sendRequest(request);
114
115 if (response.isEmpty()) {
116 taskLabel->setText("Ошибка: нет ответа от сервера.");
117 return;
118 }
119
120 // Protocol: "graph||x1;y1||x2;y2||..."
121 QStringList parts = response.trimmed().split("||");
122 if (parts.isEmpty() || parts[0] != "graph") {
123 taskLabel->setText("Ошибка: неверный формат ответа сервера.");
124 return;
125 }
126
127 points.clear();
128 for (int i = 1; i < parts.size(); ++i) {
129 QStringList xy = parts[i].split(";");
130 if (xy.size() == 2) {
131 bool okX = false, okY = false;
132 double x = xy[0].toDouble(&okX);
133 double y = xy[1].toDouble(&okY);
134 if (okX && okY)
135 points.append(QPointF(x, y));
136 }
137 }
138
139 taskLabel->setText(QString("Построено %1 точек.").arg(points.size()));
140 update(); // trigger repaint
141}

158void GraphWidget::paintEvent(QPaintEvent *event)
159{
160 QWidget::paintEvent(event);
161
162 QPainter painter(this);
163 painter.setRenderHint(QPainter::Antialiasing);
164
165 // Drawing rectangle (leave margin for labels)
166 QRect widgetRect = rect();
167
168 // Reserve space for top controls (~80px) and task label (~30px)
169 int topOffset = 90;
170 QRect drawRect(
171 MARGIN,
172 topOffset + MARGIN / 2,
173 widgetRect.width() - 2 * MARGIN,
174 widgetRect.height() - topOffset - MARGIN
175 );
176
177 if (drawRect.width() < 10 || drawRect.height() < 10)
178 return;
179
180 // Background
181 painter.fillRect(drawRect, QColor(250, 250, 255));
182
183 // --- Determine data range ---
184 double xMin = -5.0, xMax = 5.0, yMin = -3.0, yMax = 10.0;
185
186 if (!points.isEmpty()) {
187 xMin = points[0].x(); xMax = xMin;
188 yMin = points[0].y(); yMax = yMin;
189 for (const QPointF &p : points) {
190 xMin = std::min(xMin, p.x());
191 xMax = std::max(xMax, p.x());
192 yMin = std::min(yMin, p.y());
193 yMax = std::max(yMax, p.y());
194 }
195 // Add some padding to the data range
196 double xPad = (xMax - xMin) * 0.05;
197 double yPad = (yMax - yMin) * 0.05;
198 if (xPad == 0) xPad = 0.5;
199 if (yPad == 0) yPad = 0.5;
200 xMin -= xPad; xMax += xPad;
201 yMin -= yPad; yMax += yPad;
202 }
203
204 double xRange = xMax - xMin;
205 double yRange = yMax - yMin;
206 if (xRange == 0) xRange = 1;
207 if (yRange == 0) yRange = 1;
208
209 // Helpers: convert data coords -> widget coords
210 auto toWidgetX = [&](double x) -> double {
211 return drawRect.left() + (x - xMin) / xRange * drawRect.width();
212 };
213 auto toWidgetY = [&](double y) -> double {
214 return drawRect.bottom() - (y - yMin) / yRange * drawRect.height();
215 };
216
217 // --- Grid lines ---
218 painter.setPen(QPen(QColor(210, 210, 210), 1, Qt::DashLine));
219
220 // Determine nice grid step
221 auto niceStep = [](double range) -> double {
222 double raw = range / 8.0;
223 double mag = std::pow(10.0, std::floor(std::log10(raw)));
224 double norm = raw / mag;
225 double step;
226 if (norm < 1.5) step = 1;
227 else if (norm < 3.5) step = 2;
228 else if (norm < 7.5) step = 5;
229 else step = 10;
230 return step * mag;
231 };
232
233 double xStep = niceStep(xRange);
234 double yStep = niceStep(yRange);
235
236 double xStart = std::ceil(xMin / xStep) * xStep;
237 double yStart = std::ceil(yMin / yStep) * yStep;
238
239 for (double x = xStart; x <= xMax + xStep * 0.01; x += xStep) {
240 int wx = static_cast<int>(toWidgetX(x));
241 painter.drawLine(wx, drawRect.top(), wx, drawRect.bottom());
242 }
243 for (double y = yStart; y <= yMax + yStep * 0.01; y += yStep) {
244 int wy = static_cast<int>(toWidgetY(y));
245 painter.drawLine(drawRect.left(), wy, drawRect.right(), wy);
246 }
247
248 // --- Border ---
249 painter.setPen(QPen(Qt::black, 1));
250 painter.drawRect(drawRect);
251
252 // --- Axis labels ---
253 QFont smallFont = painter.font();
254 smallFont.setPointSize(8);
255 painter.setFont(smallFont);
256 painter.setPen(Qt::black);
257
258 for (double x = xStart; x <= xMax + xStep * 0.01; x += xStep) {
259 int wx = static_cast<int>(toWidgetX(x));
260 // Tick
261 painter.drawLine(wx, drawRect.bottom(), wx, drawRect.bottom() + 4);
262 QString label = QString::number(x, 'g', 4);
263 QRect textRect(wx - 25, drawRect.bottom() + 5, 50, 18);
264 painter.drawText(textRect, Qt::AlignCenter, label);
265 }
266 for (double y = yStart; y <= yMax + yStep * 0.01; y += yStep) {
267 int wy = static_cast<int>(toWidgetY(y));
268 painter.drawLine(drawRect.left() - 4, wy, drawRect.left(), wy);
269 QString label = QString::number(y, 'g', 4);
270 QRect textRect(drawRect.left() - MARGIN + 2, wy - 9, MARGIN - 8, 18);
271 painter.drawText(textRect, Qt::AlignRight | Qt::AlignVCenter, label);
272 }
273
274 // --- Coordinate axes (if visible in range) ---
275 painter.setPen(QPen(Qt::black, 2));
276
277 // X axis
278 if (yMin <= 0 && 0 <= yMax) {
279 int wy0 = static_cast<int>(toWidgetY(0));
280 painter.drawLine(drawRect.left(), wy0, drawRect.right(), wy0);
281 // Arrow
282 painter.drawLine(drawRect.right(), wy0,
283 drawRect.right() - 8, wy0 - 4);
284 painter.drawLine(drawRect.right(), wy0,
285 drawRect.right() - 8, wy0 + 4);
286 // Label
287 painter.drawText(drawRect.right() + 4, wy0 + 5, "X");
288 }
289 // Y axis
290 if (xMin <= 0 && 0 <= xMax) {
291 int wx0 = static_cast<int>(toWidgetX(0));
292 painter.drawLine(wx0, drawRect.bottom(), wx0, drawRect.top());
293 // Arrow
294 painter.drawLine(wx0, drawRect.top(),
295 wx0 - 4, drawRect.top() + 8);
296 painter.drawLine(wx0, drawRect.top(),
297 wx0 + 4, drawRect.top() + 8);
298 // Label
299 painter.drawText(wx0 + 4, drawRect.top() - 4, "Y");
300 }
301
302 // --- Function title ---
303 QFont titleFont = painter.font();
304 titleFont.setBold(true);
305 titleFont.setPointSize(9);
306 painter.setFont(titleFont);
307 painter.setPen(QColor(60, 60, 60));
308 painter.drawText(drawRect.left(), drawRect.top() - 6,
309 "f(x) = { |x|−2, x<−2 } { x²+x+1, −2≤x<2 } { |x−2|+1, x≥2 }");
310
311 // --- Graph line ---
312 if (points.size() >= 2) {
313 painter.setPen(QPen(QColor(0, 80, 220), 2));
314
315 for (int i = 1; i < points.size(); ++i) {
316 double x1 = points[i - 1].x(), y1 = points[i - 1].y();
317 double x2 = points[i].x(), y2 = points[i].y();
318 int wx1 = static_cast<int>(toWidgetX(x1));
319 int wy1 = static_cast<int>(toWidgetY(y1));
320 int wx2 = static_cast<int>(toWidgetX(x2));
321 int wy2 = static_cast<int>(toWidgetY(y2));
322 painter.drawLine(wx1, wy1, wx2, wy2);
323 }
324 } else if (points.isEmpty()) {
325 // Hint
326 QFont hintFont = painter.font();
327 hintFont.setItalic(true);
328 hintFont.setPointSize(11);
329 painter.setFont(hintFont);
330 painter.setPen(QColor(150, 150, 150));
331 painter.drawText(drawRect, Qt::AlignCenter,
332 "Нажмите «Построить график» для отображения функции");
333 }
