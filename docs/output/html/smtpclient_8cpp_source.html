14// ── Read email.txt next to the server executable ────────────────────────────
15// Format:
16// line 1: # comment / link
17// line 2: email=some@gmail.com
18// line 3: key=t q o a x w j p q l k p h a j y (spaces are stripped)
19static QString readEmailTxtValue(const QString &key)
20{
21 // Look for email.txt next to the executable, and also one level up (build dir)
22 QStringList searchPaths = {
23 QCoreApplication::applicationDirPath() + "/email.txt",
24 QCoreApplication::applicationDirPath() + "/../email.txt",
25 QDir::currentPath() + "/email.txt"
26 };
27
28 for (const QString &path : searchPaths) {
29 QFile f(path);
30 if (!f.open(QIODevice::ReadOnly | QIODevice::Text))
31 continue;
32 QTextStream in(&f);
33 while (!in.atEnd()) {
34 QString line = in.readLine();
35 if (line.startsWith('#')) continue; // skip comment lines
36 int sep = line.indexOf('=');
37 if (sep < 0) continue;
38 QString k = line.left(sep).trimmed();
39 QString v = line.mid(sep + 1).trimmed();
40 if (k == key) {
41 v.remove(' '); // strip spaces from value
42 return v;
43 }
44 }
45 }
46 qWarning() << "[SmtpClient] email.txt: key '" << key << "' not found";
47 return {};
48}

55static QByteArray waitForData(QSslSocket *socket, int timeoutMs = 15000)

68static QString readResponse(QSslSocket *socket, int timeoutMs = 15000)
69{
70 QByteArray data = waitForData(socket, timeoutMs);

91static bool doSend(const QString &toEmail,
92 const QString &subjectText,
93 const QString &bodyText)
94{
95 QSslSocket socket;
96 socket.connectToHostEncrypted(SmtpClient::smtpHost, SmtpClient::smtpPort);
97 socket.ignoreSslErrors();
98 if (!socket.waitForEncrypted(20000)) {
99 qDebug() << "[SMTP] SSL connection failed:" << socket.errorString();
100 return false;
101 }
102 qDebug() << "[SMTP] SSL connected";
103
104 QString resp = readResponse(&socket, 15000);
105 if (resp.isEmpty() || !resp.startsWith("220")) return false;
106
107 resp = sendCmd(&socket, "EHLO localhost\r\n");
108 if (!resp.startsWith("250")) return false;
109
110 resp = sendCmd(&socket, "AUTH LOGIN\r\n");
111 if (!resp.startsWith("334")) return false;
112
113 resp = sendCmd(&socket, SmtpClient::senderEmail.toUtf8().toBase64() + "\r\n");
114 if (!resp.startsWith("334")) return false;
115
116 resp = sendCmd(&socket, SmtpClient::senderPassword.toUtf8().toBase64() + "\r\n");
117 if (!resp.startsWith("235")) return false;
118 qDebug() << "[SMTP] Authenticated";
119
120 resp = sendCmd(&socket, "MAIL FROM:<" + SmtpClient::senderEmail.toUtf8() + ">\r\n");
121 if (!resp.startsWith("250")) return false;
122
123 resp = sendCmd(&socket, "RCPT TO:<" + toEmail.toUtf8() + ">\r\n");
124 if (!resp.startsWith("250")) return false;
125
126 resp = sendCmd(&socket, "DATA\r\n");
127 if (!resp.startsWith("354")) return false;
128
129 QByteArray subjectB64 = subjectText.toUtf8().toBase64();
130
131 QByteArray email;
132 email += "Subject: =?UTF-8?B?" + subjectB64 + "?=\r\n";
133 email += "From: " + SmtpClient::senderEmail.toUtf8() + "\r\n";
134 email += "To: " + toEmail.toUtf8() + "\r\n";
135 email += "Content-Type: text/plain; charset=UTF-8\r\n";
136 email += "\r\n";
137 email += bodyText.toUtf8() + "\r\n";
138 email += ".\r\n";
139
140 socket.write(email);
141 socket.flush();
142 socket.waitForBytesWritten(5000);
143 resp = readResponse(&socket);
144 if (!resp.startsWith("250")) {
145 qDebug() << "[SMTP] Email rejected:" << resp;
146 return false;
147 }
148
149 sendCmd(&socket, "QUIT\r\n");
150 socket.disconnectFromHost();
151 if (socket.state() != QAbstractSocket::UnconnectedState)
152 socket.waitForDisconnected(5000);
153
154 qDebug() << "[SMTP] Email sent to" << toEmail;
155 return true;
156}
